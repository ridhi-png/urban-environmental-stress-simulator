16@st.cache_data 

17def load_data() -> pd.DataFrame: 

18 """ 

19 Load environmental data from CSV and calculate initial stress metrics. 

20  

21 This function loads the city environmental data from the CSV file, calculates 

22 stress scores using the weighted formula, and classifies risk levels for each zone. 

23 The @st.cache_data decorator ensures the data is loaded only once and cached 

24 for performance, avoiding redundant calculations on every user interaction. 

25  

26 Data Processing Steps: 

27 1. Load CSV file from data/city_environment.csv 

28 2. Calculate stress scores using the stress engine (weighted normalization) 

29 3. Classify risk levels (Low, Moderate, High) based on stress scores 

30  

31 Returns: 

32 pd.DataFrame: DataFrame with columns: 

33 - zone: Zone identifier (Zone A through Zone J) 

34 - AQI: Air Quality Index 

35 - waste_index: Waste management metric 

36 - temperature: Temperature in Celsius 

37 - humidity: Relative humidity percentage 

38 - stress_score: Calculated composite stress score (0-1 range) 

39 - risk_level: Risk classification (Low, Moderate, High) 

40  

41 Raises: 

42 FileNotFoundError: If data/city_environment.csv does not exist 

43 pd.errors.EmptyDataError: If the CSV file is empty 

44 KeyError: If required columns are missing from the CSV 

45  

46 Examples: 

47 >>> df = load_data() 

48 >>> len(df) 

49 10 

50 >>> 'stress_score' in df.columns and 'risk_level' in df.columns 

51 True 

52 """ 

53 # Define the path to the CSV file 

54 csv_path = Path("data/city_environment.csv") 

55 

56 # Check if the file exists and provide helpful error message 

57 if not csv_path.exists(): 

58 raise FileNotFoundError( 

59 f"Data file not found at {csv_path}. " 

60 "Please run the data generation script first: python generate_data.py" 

61 ) 

62 

63 try: 

64 # Load the CSV file into a pandas DataFrame 

65 df = pd.read_csv(csv_path) 

66 

67 # Validate that the dataframe is not empty 

68 if df.empty: 

69 raise pd.errors.EmptyDataError("The CSV file is empty. Please regenerate the data.") 

70 

71 # Validate that required columns exist 

72 required_columns = ['zone', 'AQI', 'waste_index', 'temperature', 'humidity'] 

73 missing_columns = [col for col in required_columns if col not in df.columns] 

74 if missing_columns: 

75 raise KeyError( 

76 f"Missing required columns: {missing_columns}. " 

77 "Please ensure the CSV has all required columns." 

78 ) 

79 

80 # Calculate initial stress scores using the stress engine 

81 # This applies normalization and weighted formula to AQI, waste_index, and temperature 

82 df = calculate_stress_score(df) 

83 

84 # Classify risk levels for each zone based on stress scores 

85 df['risk_level'] = df['stress_score'].apply(classify_risk) 

86 

87 return df 

88 

89 except pd.errors.EmptyDataError as e: 

90 raise pd.errors.EmptyDataError(f"Error reading CSV file: {e}") 

91 except Exception as e: 

92 raise Exception(f"Unexpected error loading data: {e}") 

93
