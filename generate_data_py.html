2Data Generation Script

4Generates synthetic environmental data for 10 urban zones. 

5Creates a CSV file with realistic varied environmental metrics.

7 

8import random 

9import csv 

10import os

12 

13def generate_city_environment_data(): 

14 """ 

15 Generate synthetic environmental data for 10 urban zones. 

16  

17 This function creates realistic synthetic data for testing and demonstrating 

18 the Urban Environmental Stress Simulator. The data represents environmental 

19 conditions across 10 urban zones with varied pollution, waste, temperature, 

20 and humidity levels. 

21  

22 Data Generation Strategy: 

23 - Uses Python's random module to generate varied values within realistic ranges 

24 - Each zone gets independent random values to simulate diverse urban conditions 

25 - Ranges are chosen to represent typical urban environmental conditions: 

26 * AQI (50-300): Covers "Moderate" to "Hazardous" air quality levels 

27 * Waste Index (20-90): Represents varying waste management effectiveness 

28 * Temperature (15-40°C): Covers temperate to hot urban climates 

29 * Humidity (30-90%): Represents dry to humid conditions 

30  

31 Creates data/city_environment.csv with the following columns: 

32 - zone: Zone identifier (Zone A through Zone J) 

33 - AQI: Air Quality Index (50-300) 

34 - waste_index: Waste management metric (20-90) 

35 - temperature: Temperature in Celsius (15-40) 

36 - humidity: Relative humidity percentage (30-90) 

37  

38 Returns: 

39 str: Path to the generated CSV file 

40  

41 Requirements: 1.1, 1.2, 1.3, 1.4, 1.5, 1.6, 1.7 

42 """ 

43 # Define zone labels using ASCII character codes 

44 # chr(65) = 'A', chr(66) = 'B', etc. 

45 # This generates: ["Zone A", "Zone B", ..., "Zone J"] 

46 zones = [f"Zone {chr(65 + i)}" for i in range(10)] # Zone A through Zone J 

47 

48 # Generate random environmental data for each zone 

49 data = [] 

50 for zone in zones: 

51 # Create a dictionary for each zone with randomized environmental metrics 

52 row = { 

53 'zone': zone, 

54 # AQI range (50-300): Represents air pollution levels 

55 # Lower values = better air quality, higher values = worse air quality 

56 'AQI': random.randint(50, 300), 

57 

58 # Waste index range (20-90): Represents waste management effectiveness 

59 # Lower values = better waste management, higher values = more waste issues 

60 'waste_index': random.randint(20, 90), 

61 

62 # Temperature range (15-40°C): Represents ambient temperature 

63 # Rounded to 1 decimal place for realistic precision 

64 'temperature': round(random.uniform(15, 40), 1), 

65 

66 # Humidity range (30-90%): Represents relative humidity 

67 # Integer values are sufficient for humidity measurements 

68 'humidity': random.randint(30, 90) 

69 } 

70 data.append(row) 

71 

72 # Ensure the data directory exists before writing the file 

73 # exist_ok=True prevents errors if the directory already exists 

74 os.makedirs('data', exist_ok=True) 

75 

76 # Write the generated data to a CSV file 

77 csv_path = 'data/city_environment.csv' 

78 with open(csv_path, 'w', newline='') as csvfile: 

79 # Define the column order for the CSV file 

80 fieldnames = ['zone', 'AQI', 'waste_index', 'temperature', 'humidity'] 

81 

82 # Create a CSV DictWriter to write dictionaries as CSV rows 

83 writer = csv.DictWriter(csvfile, fieldnames=fieldnames) 

84 

85 # Write the header row with column names 

86 writer.writeheader() 

87 

88 # Write all data rows 

89 writer.writerows(data) 

90 

91 # Print confirmation messages 

92 print(f"✓ Generated environmental data for {len(data)} zones") 

93 print(f"✓ Saved to {csv_path}") 

94 return csv_path
